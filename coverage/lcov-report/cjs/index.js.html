function done(err) { if (arguments.length) throw err; }
function isPromise(value) { return value != null && typeof value.then === 'function'; }
function unshiftThen(queue, value, resolve) {
  if (isPromise(value)) queue.unshift(value);
  resolve();
}
 
module.exports = function (options) { if (!options) options = {};
  var after = options.done || done;
  var next = options.next || function (value) { return value; };
  var queue = [];
  var status = 'initialized';
  return {
    get state() { return status; },
    by: function () {
      queue.push.apply(queue, arguments);
      if (status !== 'pending') {
        status = 'pending';
        (function resolve() {
          if (queue.length) {
            const result = queue.shift();
            if (isPromise(result)) {
              result.then(
                function (value) { unshiftThen(queue, next(value), resolve) },
                function (error) {
                  queue.splice(0);
                  status = 'rejected';
                  after(error);
                }
              );
            } else
              unshiftThen(queue, next(result), resolve);
          } else {
            status = 'resolved';
            after();
          }
        }());
      }
      return this;
    }
  };
};
